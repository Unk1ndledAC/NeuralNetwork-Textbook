\chapter{Parameter and Computation Analysis}

\section{Importance of Model Efficiency}

Understanding model efficiency is crucial for:
\begin{itemize}
    \item Deployment on resource-constrained devices
    \item Training cost estimation
    \item Model selection and architecture design
    \item Benchmarking and comparison
\end{itemize}

Key metrics:
\begin{itemize}
    \item \textbf{Parameters:} Model size and memory footprint
    \item \textbf{FLOPs:} Computational complexity
    \item \textbf{Latency:} Inference time
    \item \textbf{Throughput:} Samples per second
\end{itemize}

\section{Parameter Counting}

Parameters are the learnable weights and biases in the network.

\subsection{Fully Connected Layer}

For input dimension $d_{in}$ and output dimension $d_{out}$:
\begin{align}
\text{Weights} &= d_{in} \times d_{out} \\
\text{Biases} &= d_{out} \\
\text{Total} &= d_{in} \times d_{out} + d_{out}
\end{align}

Example: FC layer from 784 to 256:
\begin{equation}
784 \times 256 + 256 = 200,704 + 256 = 200,960 \text{ parameters}
\end{equation}

\subsection{Convolutional Layer}

For input channels $C_{in}$, output channels $C_{out}$, kernel size $K$:
\begin{align}
\text{Weights} &= K \times K \times C_{in} \times C_{out} \\
\text{Biases} &= C_{out} \quad \text{(typically)} \\
\text{Total} &= K^2 \times C_{in} \times C_{out} + C_{out}
\end{align}

Example: Conv layer $3\times3$, 64 in, 128 out:
\begin{equation}
3 \times 3 \times 64 \times 128 + 128 = 73,728 + 128 = 73,856 \text{ parameters}
\end{equation}

\subsection{Batch Normalization}

For $C$ channels:
\begin{align}
\gamma \text{ (scale)} &= C \\
\beta \text{ (shift)} &= C \\
\mu \text{ (running mean)} &= C \quad \text{(inference only)} \\
\sigma \text{ (running var)} &= C \quad \text{(inference only)} \\
\text{Total trainable} &= 2C
\end{align}

Note: During training, BN has 2 trainable and 2 non-trainable parameters per channel.

\section{FLOPs (Floating Point Operations)}

FLOPs measure the computational cost of a forward pass.

\subsection{FLOPs for FC Layer}

For matrix multiplication $\mathbf{Y} = \mathbf{W}\mathbf{X} + \mathbf{b}$:
\begin{align}
\text{Multiplications} &= d_{out} \times d_{in} \quad \text{(per sample)} \\
\text{Additions} &= d_{out} \times (d_{in} - 1) \quad \text{(approximately } d_{out} \times d_{in}) \\
\text{FLOPs} &\approx 2 \times d_{out} \times d_{in} \quad \text{(counting mult+add as 2 FLOPs)}
\end{align}

For batch of $N$: FLOPs $= 2 \times N \times d_{out} \times d_{in}$

\subsection{FLOPs for Conv Layer}

For output feature map $H_{out} \times W_{out}$:
\begin{align}
\text{FLOPs per output pixel} &= K^2 \times C_{in} \times C_{out} \quad \text{(mult-add pairs)} \times 2 \\
\text{Total FLOPs} &= H_{out} \times W_{out} \times K^2 \times C_{in} \times C_{out} \times 2
\end{align}

Example: $224\times224$ input, $3\times3$ conv, 64 in, 128 out:
\begin{align}
H_{out} \times W_{out} &= 224 \times 224 = 50,176 \\
\text{FLOPs} &= 50,176 \times 3 \times 3 \times 64 \times 128 \times 2 \approx 1.17 \text{ GFLOPs}
\end{align}

\begin{note}[MACs vs FLOPs]
Sometimes FLOPs are reported as Multiply-Accumulate (MAC) operations:
\begin{equation}
\text{MAC} = \sum_{i} a_i \cdot b_i
\end{equation}
Each MAC typically counts as 1 operation (or 2 FLOPs if counting mult and add separately).

1 MAC $\approx$ 2 FLOPs for basic arithmetic.
\end{note}

\section{Transfer Learning and Efficiency}

Transfer learning can dramatically reduce required data and training.

\begin{properties}[Transfer Learning Benefits]
\begin{itemize}
    \item Lower training cost (no training from scratch)
    \item Can use smaller datasets
    \item Faster convergence
    \item Often better final performance
\end{itemize}
\end{properties}

Common strategies:
\begin{enumerate}
    \item \textbf{Freeze:} Keep pretrained weights fixed
    \item \textbf{Fine-tune:} Adapt all/some layers with small learning rate
    \item \textbf{Extract features:} Use pretrained as feature extractor
\end{enumerate}

\subsection{Transformer Model Scaling}

For a transformer model with $L$ layers, $d_{model}$ hidden dimension, $h$ attention heads, and vocabulary size $V$:

\begin{align}
\text{Embedding params} &= V \times d_{model} \\
\text{Per-layer params} &\approx 12 \cdot d_{model}^2 \quad \text{(attention: } 4d_{model}^2 \text{, FFN: } 8d_{model}^2) \\
\text{Total params} &\approx V \cdot d_{model} + 12 L \cdot d_{model}^2
\end{align}

\begin{note}[The Scaling Laws]
The scaling-law analysis in \cite{kaplan2020scaling} showed that transformer performance depends predictably on three factors:
\begin{equation}
\mathcal{L}(N, D, C) \approx \left(\frac{N_c}{N}\right)^{\alpha_N} + \left(\frac{D_c}{D}\right)^{\alpha_D}
\end{equation}
where $N$ = parameters, $D$ = data size, $C$ = compute, and $\alpha_N \approx 0.076$, $\alpha_D \approx 0.095$. This implies that to maintain efficiency, parameters, data, and compute should scale proportionally ($\approx \times 10$ each for each ``order of magnitude'' improvement).
\end{note}

\section{Activation Memory Analysis}

During training, memory consumption is dominated by \textbf{activations} (intermediate feature maps stored for backpropagation), not parameters.

\subsection{Activation Memory Formula}

For a convolutional layer output of shape $(N, C, H, W)$ in FP32:
\begin{equation}
\text{Activation memory} = N \times C \times H \times W \times 4 \text{ bytes (FP32)}
\end{equation}

\begin{example}[ResNet-50 Activation Memory]
For batch size $N = 32$, FP32:
\begin{itemize}
    \item Stage 1 output ($56^2 \times 256$): $32 \times 256 \times 56^2 \times 4 = 103$ MB
    \item Stage 2 output ($28^2 \times 512$): $32 \times 512 \times 28^2 \times 4 = 51$ MB
    \item Stage 3 output ($14^2 \times 1024$): $32 \times 1024 \times 14^2 \times 4 = 26$ MB
    \item Total activations for all layers (including intermediates): $\approx 3$--$4$ GB
\end{itemize}
With mixed precision (FP16), activation memory is halved ($\approx 2$ GB).
\end{example}

\begin{properties}[Reducing Activation Memory]
\begin{itemize}
    \item \textbf{Gradient checkpointing}: Recompute activations during backward pass. Trades compute ($\sim$33\% overhead) for memory ($\sim$60\% reduction).
    \item \textbf{Mixed precision (AMP)}: Store activations in FP16, master weights in FP32. Halves activation memory with minimal accuracy loss.
    \item \textbf{Gradient accumulation}: Simulate large batch sizes without proportional memory increase.
    \item \textbf{DeepSpeed ZeRO}: Partition optimizer states, gradients, and parameters across GPUs.
\end{itemize}
\end{properties}

\section{Model Compression and Quantization}

\subsection{Quantization}

Quantization reduces the numerical precision of weights and activations:

\begin{table}[ht]
    \centering
    \caption{Quantization Levels}
    \begin{tabular}{L{2cm} L{2.5cm} L{2.5cm} L{2.5cm} L{2.5cm}}
        \toprule
        \textbf{Precision} & \textbf{Bytes/param} & \textbf{Range} & \textbf{Size (100M)} & \textbf{Speedup} \\
        \midrule
        FP32 & 4 & $\pm 3.4\times10^{38}$ & 400 MB & 1$\times$ \\
        FP16 & 2 & $\pm 65504$ & 200 MB & 2$\times$ \\
        BF16 & 2 & $\pm 3.4\times10^{38}$ & 200 MB & 2$\times$ \\
        INT8 & 1 & $[-128, 127]$ & 100 MB & 2--4$\times$ \\
        INT4 & 0.5 & $[-8, 7]$ & 50 MB & 4--8$\times$ \\
        \bottomrule
    \end{tabular}
\end{table}

\textbf{Post-training quantization (PTQ)} quantizes a trained model without retraining. \textbf{Quantization-aware training (QAT)} simulates quantization during training for higher accuracy.

\subsection{Other Compression Techniques}

\begin{itemize}
    \item \textbf{Pruning}: Remove unimportant weights (magnitude-based or structured). Can reduce FLOPs by 2--3$\times$ with $<1\%$ accuracy loss.
    \item \textbf{Knowledge distillation}: Train a smaller ``student'' model to mimic a larger ``teacher''. The student often exceeds a same-size model trained directly.
    \item \textbf{Low-rank factorization}: Decompose weight matrices $\mathbf{W} = \mathbf{U}\mathbf{V}^\top$ where $\text{rank}(\mathbf{U}) = \text{rank}(\mathbf{V}) \ll \text{rank}(\mathbf{W})$.
\end{itemize}

\section{Chapter Summary}

\begin{enumerate}
    \item Parameter count determines model size and memory requirements
    \item FLOPs measure computational complexity (1 MAC $\approx$ 2 FLOPs)
    \item ResNet-50: 25.6M params, 4.1B FLOPs; transformer params scale as $12Ld^2$
    \item Activation memory often exceeds parameter memory during training
    \item Gradient checkpointing and mixed precision reduce memory with compute tradeoffs
    \item Quantization (INT8, INT4) enables efficient deployment on edge devices
    \item Transfer learning leverages pretrained models efficiently
    \item Scaling laws guide the design of large foundation models
\end{enumerate}
