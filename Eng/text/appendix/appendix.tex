\chapter{Mathematical Reference}

\section{Linear Algebra}

\subsection{Matrix Operations}

\begin{itemize}
    \item $\mathbf{AB} \neq \mathbf{BA}$ (non-commutative in general)
    \item $(\mathbf{AB})^\top = \mathbf{B}^\top\mathbf{A}^\top$
    \item $(\mathbf{AB})^{-1} = \mathbf{B}^{-1}\mathbf{A}^{-1}$
    \item $\det(\mathbf{AB}) = \det(\mathbf{A})\det(\mathbf{B})$
    \item $\text{tr}(\mathbf{AB}) = \text{tr}(\mathbf{BA})$
\end{itemize}

\subsection{Norms}

For a vector $\mathbf{x} \in \mathbb{R}^n$:
\begin{align}
\|\mathbf{x}\|_1 &= \sum_{i=1}^{n} |x_i| \quad \text{(L1 norm)} \\
\|\mathbf{x}\|_2 &= \sqrt{\sum_{i=1}^{n} x_i^2} \quad \text{(L2 / Euclidean norm)} \\
\|\mathbf{x}\|_p &= \left(\sum_{i=1}^{n} |x_i|^p\right)^{1/p} \quad \text{(L$p$ norm)} \\
\|\mathbf{x}\|_\infty &= \max_i |x_i| \quad \text{(L$\infty$ norm)}
\end{align}

For a matrix $\mathbf{A}$:
\begin{align}
\|\mathbf{A}\|_F &= \sqrt{\sum_{i,j} A_{ij}^2} \quad \text{(Frobenius norm)} \\
\|\mathbf{A}\|_2 &= \sigma_{\max}(\mathbf{A}) \quad \text{(Spectral norm = largest singular value)}
\end{align}

\subsection{Eigenvalue Decomposition}

A square matrix $\mathbf{A} \in \mathbb{R}^{n \times n}$ can be decomposed (if diagonalizable):
\begin{equation}
\mathbf{A} = \mathbf{Q}\boldsymbol{\Lambda}\mathbf{Q}^{-1}
\end{equation}
where $\boldsymbol{\Lambda} = \text{diag}(\lambda_1, \ldots, \lambda_n)$ are eigenvalues and $\mathbf{Q}$ contains eigenvectors.

\textbf{Properties:}
\begin{itemize}
    \item $\det(\mathbf{A}) = \prod_{i=1}^{n} \lambda_i$
    \item $\text{tr}(\mathbf{A}) = \sum_{i=1}^{n} \lambda_i$
    \item $\mathbf{A}$ is positive definite iff $\lambda_i > 0$ for all $i$
    \item $\mathbf{A}^k = \mathbf{Q}\boldsymbol{\Lambda}^k\mathbf{Q}^{-1}$
\end{itemize}

\subsection{Singular Value Decomposition (SVD)}

Any matrix $\mathbf{A} \in \mathbb{R}^{m \times n}$ can be decomposed:
\begin{equation}
\mathbf{A} = \mathbf{U}\boldsymbol{\Sigma}\mathbf{V}^\top
\end{equation}
where $\mathbf{U} \in \mathbb{R}^{m \times m}$ (left singular vectors), $\boldsymbol{\Sigma} \in \mathbb{R}^{m \times n}$ (diagonal singular values $\sigma_1 \geq \sigma_2 \geq \cdots \geq 0$), $\mathbf{V} \in \mathbb{R}^{n \times n}$ (right singular vectors).

\textbf{Applications in deep learning:}
\begin{itemize}
    \item Low-rank approximation: $\mathbf{A}_k = \mathbf{U}_k \boldsymbol{\Sigma}_k \mathbf{V}_k^\top$ (keep top $k$ singular values)
    \item Principal Component Analysis (PCA)
    \item Understanding weight matrices: $\sigma_{\max}(\mathbf{W})$ governs gradient flow through a layer
\end{itemize}

\section{Calculus}

\subsection{Common Derivatives}

\begin{align}
\frac{d}{dx}x^n &= nx^{n-1} \\
\frac{d}{dx}e^x &= e^x \\
\frac{d}{dx}\log(x) &= \frac{1}{x} \\
\frac{d}{dx}\sin(x) &= \cos(x) \\
\frac{d}{dx}\cos(x) &= -\sin(x) \\
\frac{d}{dx}\sigma(x) &= \sigma(x)(1 - \sigma(x)) \quad \text{(sigmoid)} \\
\frac{d}{dx}\text{ReLU}(x) &= \begin{cases} 1 & x > 0 \\ 0 & x \leq 0 \end{cases} \\
\frac{d}{dx}\tanh(x) &= 1 - \tanh^2(x)
\end{align}

\subsection{Chain Rule}

For $y = f(g(x))$:
\begin{equation}
\frac{dy}{dx} = f'(g(x)) \cdot g'(x)
\end{equation}

For multivariable functions, the chain rule becomes:
\begin{equation}
\frac{\partial \mathcal{L}}{\partial \mathbf{x}} = \frac{\partial \mathcal{L}}{\partial \mathbf{y}} \cdot \frac{\partial \mathbf{y}}{\partial \mathbf{x}}
\end{equation}
where $\frac{\partial \mathbf{y}}{\partial \mathbf{x}}$ is the \textbf{Jacobian matrix} with entries $J_{ij} = \frac{\partial y_i}{\partial x_j}$.

\subsection{Jacobian and Hessian}

\textbf{Jacobian} of $\mathbf{f}: \mathbb{R}^n \to \mathbb{R}^m$ at $\mathbf{x}$:
\begin{equation}
\mathbf{J}_f(\mathbf{x}) = \begin{bmatrix}
\frac{\partial f_1}{\partial x_1} & \cdots & \frac{\partial f_1}{\partial x_n} \\
\vdots & \ddots & \vdots \\
\frac{\partial f_m}{\partial x_1} & \cdots & \frac{\partial f_m}{\partial x_n}
\end{bmatrix} \in \mathbb{R}^{m \times n}
\end{equation}

\textbf{Hessian} of $f: \mathbb{R}^n \to \mathbb{R}$ at $\mathbf{x}$:
\begin{equation}
\mathbf{H}_f(\mathbf{x}) = \begin{bmatrix}
\frac{\partial^2 f}{\partial x_1^2} & \cdots & \frac{\partial^2 f}{\partial x_1 \partial x_n} \\
\vdots & \ddots & \vdots \\
\frac{\partial^2 f}{\partial x_n \partial x_1} & \cdots & \frac{\partial^2 f}{\partial x_n^2}
\end{bmatrix} \in \mathbb{R}^{n \times n}
\end{equation}

\textbf{Uses in deep learning:}
\begin{itemize}
    \item Jacobians describe gradient flow through layers (critical for RNN vanishing gradient analysis)
    \item The Hessian's eigenvalues determine the loss landscape curvature (sharp vs flat minima)
    \item Second-order methods (Newton's method) use $\mathbf{H}^{-1}$ for parameter updates
\end{itemize}

\subsection{Taylor Expansion}

For a smooth function $f(\mathbf{x})$ around $\mathbf{x}_0$:
\begin{equation}
f(\mathbf{x}) = f(\mathbf{x}_0) + \nabla f(\mathbf{x}_0)^\top(\mathbf{x} - \mathbf{x}_0) + \frac{1}{2}(\mathbf{x} - \mathbf{x}_0)^\top \mathbf{H}(\mathbf{x}_0)(\mathbf{x} - \mathbf{x}_0) + \cdots
\end{equation}

\subsection{Lagrange Multipliers}

For constrained optimization $\min_{\mathbf{x}} f(\mathbf{x})$ subject to $g(\mathbf{x}) = 0$:
\begin{equation}
\mathcal{L}(\mathbf{x}, \lambda) = f(\mathbf{x}) + \lambda \cdot g(\mathbf{x})
\end{equation}
Solution: $\nabla_\mathbf{x} \mathcal{L} = \mathbf{0}$, $\nabla_\lambda \mathcal{L} = 0$.

\section{Probability and Statistics}

\subsection{Bayes' Theorem}

\begin{equation}
P(A|B) = \frac{P(B|A)P(A)}{P(B)}
\end{equation}

\subsection{Distributions}

\begin{align}
\text{Gaussian:} \quad & p(x) = \frac{1}{\sigma\sqrt{2\pi}} \exp\left(-\frac{(x-\mu)^2}{2\sigma^2}\right) \\
\text{Bernoulli:} \quad & P(X=k) = p^k(1-p)^{1-k}, \quad k \in \{0, 1\} \\
\text{Categorical:} \quad & P(X=k) = p_k, \quad \sum_{k} p_k = 1
\end{align}

\subsection{Expectation and Variance}

\begin{align}
\mathbb{E}[X] &= \sum_x x \cdot P(X=x) \quad \text{(discrete)} \\
\mathbb{E}[X] &= \int x \cdot p(x) \, dx \quad \text{(continuous)} \\
\text{Var}(X) &= \mathbb{E}[X^2] - (\mathbb{E}[X])^2 \\
\text{Var}(aX + b) &= a^2 \text{Var}(X) \\
\mathbb{E}[XY] &= \mathbb{E}[X]\mathbb{E}[Y] \quad \text{if } X \perp Y
\end{align}

\subsection{KL Divergence and Cross-Entropy}

\begin{align}
\text{KL}(p \| q) &= \sum_x p(x)\log\frac{p(x)}{q(x)} \quad \text{(KL divergence, $\geq 0$)} \\
H(p, q) &= -\sum_x p(x)\log q(x) \quad \text{(cross-entropy)} \\
H(p, q) &= H(p) + \text{KL}(p \| q)
\end{align}

\subsection{Information Theory}

\begin{align}
H(X) &= -\sum_x P(X=x)\log P(X=x) \quad \text{(entropy)} \\
I(X; Y) &= H(X) - H(X|Y) \quad \text{(mutual information)}
\end{align}

\section{Model Creation}

\begin{lstlisting}[language=Python, caption={PyTorch model definition patterns}]
class ConvBlock(nn.Module):
    def __init__(self, in_ch, out_ch, stride=1):
        super().__init__()
        self.conv = nn.Conv2d(in_ch, out_ch, 3, stride, 1, bias=False)
        self.bn = nn.BatchNorm2d(out_ch)
        self.relu = nn.ReLU(inplace=True)

    def forward(self, x):
        return self.relu(self.bn(self.conv(x)))

class MyModel(nn.Module):
    def __init__(self, num_classes=10):
        super().__init__()
        self.features = nn.Sequential(
            ConvBlock(3, 64),
            nn.MaxPool2d(2),
            ConvBlock(64, 128),
            nn.AdaptiveAvgPool2d(1)
        )
        self.classifier = nn.Linear(128, num_classes)

    def forward(self, x):
        x = self.features(x)
        x = x.flatten(1)
        return self.classifier(x)
\end{lstlisting}

\chapter{Glossary}

\section{Architecture Terms}

\begin{description}
    \item[ANN] Artificial Neural Network --- the broadest class of neural networks
    \item[CNN] Convolutional Neural Network --- specialized for grid-like data (images, audio spectrograms)
    \item[RNN] Recurrent Neural Network --- processes sequential data with hidden state recurrence
    \item[LSTM] Long Short-Term Memory --- gated RNN that solves the vanishing gradient problem
    \item[GRU] Gated Recurrent Unit --- simplified LSTM with two gates (reset, update)
    \item[Transformer] Attention-based architecture that processes sequences in parallel
    \item[ViT] Vision Transformer --- applies the Transformer architecture to image patches
    \item[GAN] Generative Adversarial Network --- generator-discriminator framework for generative modeling
    \item[VAE] Variational Autoencoder --- probabilistic generative model with encoder-decoder structure
    \item[Autoencoder] Self-supervised model that learns to compress and reconstruct data
    \item[U-Net] Encoder-decoder architecture with skip connections for segmentation
\end{description}

\section{Training Terms}

\begin{description}
    \item[Epoch] One complete pass through the entire training dataset
    \item[Batch] A subset of samples processed together in one forward/backward pass
    \item[Batch Size] Number of samples per batch (e.g., 32, 64, 256)
    \item[Backpropagation] Algorithm for computing gradients via reverse-mode automatic differentiation
    \item[Learning Rate (LR)] Step size for parameter updates during optimization
    \item[Loss Function] Objective function that the model minimizes during training
    \item[Gradient] Partial derivative of the loss with respect to model parameters
    \item[Convergence] State where the loss stops decreasing (model has learned the data)
    \item[Overfitting] Model performs well on training data but poorly on unseen data
    \item[Generalization] Model's ability to perform well on unseen data
    \item[Regularization] Techniques to prevent overfitting (dropout, weight decay, etc.)
    \item[Early Stopping] Stop training when validation performance degrades
    \item[Fine-tuning] Adapting a pretrained model to a new task/dataset
    \item[Pre-training] Training a model on a large dataset before task-specific adaptation
    \item[Transfer Learning] Using knowledge from one task to improve another
\end{description}

\section{Layer and Operation Terms}

\begin{description}
    \item[Convolution] Operation that slides a kernel across input to extract local features
    \item[Pooling] Spatial downsampling operation (max pooling, average pooling)
    \item[Stride] Step size for the sliding window in convolution/pooling
    \item[Padding] Adding zeros (or values) around input borders
    \item[Kernel / Filter] Learnable weights applied in convolution
    \item[Feature Map] Output of a convolutional layer
    \item[Activation Function] Nonlinear function applied after linear transformation
    \item[Dropout] Randomly zeroing activations during training for regularization
    \item[BatchNorm] Batch Normalization --- normalizes features across the batch dimension
    \item[LayerNorm] Layer Normalization --- normalizes features across the feature dimension
    \item[Softmax] Converts logits to probability distribution (sums to 1)
    \item[Embedding] Dense vector representation of discrete tokens (words, patches)
    \item[Attention] Mechanism that computes weighted sums of values based on query-key similarity
    \item[Self-Attention] Attention where queries, keys, and values come from the same sequence
    \item[Cross-Attention] Attention where queries and keys/values come from different sequences
    \item[Positional Encoding] Encoding of position information for sequence models
    \item[Residual Connection] Skip connection that adds input to output ($\mathbf{y} = \mathbf{x} + \mathcal{F}(\mathbf{x})$)
\end{description}

\section{Performance Metrics}

\begin{description}
    \item[FLOPs] Floating Point Operations --- measure of computational cost
    \item[MACs] Multiply-Accumulate operations ($\approx$ half of FLOPs)
    \item[GFLOPs] Giga FLOPs ($10^9$ FLOPs)
    \item[Top-1 Accuracy] Fraction of samples where the highest-confidence prediction is correct
    \item[Top-5 Accuracy] Fraction of samples where the correct class is in the top 5 predictions
    \item[mAP] Mean Average Precision --- standard metric for object detection
    \item[IoU] Intersection over Union --- overlap measure for segmentation/detection
    \item[BLEU] Bilingual Evaluation Understudy --- metric for text generation quality
    \item[Perplexity] Exponential of cross-entropy; lower is better for language models
    \item[Throughput] Number of samples processed per second (samples/sec)
    \item[Latency] Time to process a single sample (ms)
\end{description}
