\newglossaryentry{ANN}{
    name={Artificial Neural Network (ANN)},
    description={A computing system inspired by biological neural networks that consists of interconnected nodes (neurons) organized in layers}
}

\newglossaryentry{NN}{
    name={Neural Network (NN)},
    description={See Artificial Neural Network}
}

\newglossaryentry{DNN}{
    name={Deep Neural Network (DNN)},
    description={A neural network with multiple hidden layers between input and output layers, enabling the learning of complex representations}
}

\newglossaryentry{CNN}{
    name={Convolutional Neural Network (CNN)},
    description={A deep learning architecture specifically designed for processing structured grid data like images, using convolutional layers}
}

\newglossaryentry{RNN}{
    name={Recurrent Neural Network (RNN)},
    description={A neural network architecture designed for sequential data processing, where connections between nodes form a directed sequence}
}

\newglossaryentry{LSTM}{
    name={Long Short-Term Memory (LSTM)},
    description={A type of RNN architecture that uses gating mechanisms to control information flow, addressing the vanishing gradient problem}
}

\newglossaryentry{GRU}{
    name={Gated Recurrent Unit (GRU)},
    description={A type of RNN architecture with fewer gates than LSTM, often achieving comparable performance with simpler structure}
}

\newglossaryentry{MLP}{
    name={Multi-Layer Perceptron (MLP)},
    description={A fully connected feedforward neural network consisting of an input layer, one or more hidden layers, and an output layer}
}

\newglossaryentry{BN}{
    name={Batch Normalization (BN)},
    description={A technique that normalizes layer inputs across a batch to stabilize and accelerate neural network training}
}

\newglossaryentry{Dropout}{
    name={Dropout},
    description={A regularization technique that randomly sets a fraction of neuron outputs to zero during training to prevent overfitting}
}

\newglossaryentry{FLOPs}{
    name={Floating Point Operations (FLOPs)},
    description={A measure of computational complexity representing the number of floating-point arithmetic operations performed}
}

\newglossaryentry{GPU}{
    name={Graphics Processing Unit (GPU)},
    description={A specialized electronic circuit designed to accelerate image rendering, now widely used for parallel computing in deep learning}
}

\newglossaryentry{CUDA}{
    name={Compute Unified Device Architecture (CUDA)},
    description={A parallel computing platform and API developed by NVIDIA for general-purpose computing on GPUs}
}

\newglossaryentry{backprop}{
    name={Backpropagation},
    description={An algorithm for computing gradients of the loss function with respect to network weights by applying the chain rule}
}

\newglossaryentry{SGD}{
    name={Stochastic Gradient Descent (SGD)},
    description={An optimization algorithm that uses random subsets (mini-batches) of training data to compute gradients and update weights}
}

\newglossaryentry{Adam}{
    name={Adam Optimizer},
    description={An adaptive learning rate optimization algorithm that combines momentum and RMSprop, computing individual learning rates for parameters}
}

\newglossaryentry{epoch}{
    name={Epoch},
    description={One complete pass through the entire training dataset during neural network training}
}

\newglossaryentry{batch}{
    name={Batch Size},
    description={The number of training examples processed before updating model parameters}
}

\newglossaryentry{learningrate}{
    name={Learning Rate},
    description={A hyperparameter that controls how much to change the model parameters during each update step}
}

\newglossaryentry{overfitting}{
    name={Overfitting},
    description={A phenomenon where a model learns the training data too well, including noise, resulting in poor generalization to new data}
}

\newglossaryentry{underfitting}{
    name={Underfitting},
    description={A phenomenon where a model fails to learn the underlying pattern in the training data, performing poorly on both training and test sets}
}

\newglossaryentry{regularization}{
    name={Regularization},
    description={Techniques used to prevent overfitting by adding constraints to the model or modifying the training objective}
}

\newglossaryentry{lossfunction}{
    name={Loss Function},
    description={A function that measures the difference between predicted and actual values, used to guide model optimization}
}

\newglossaryentry{activation}{
    name={Activation Function},
    description={A mathematical function applied to neuron outputs to introduce non-linearity into the network}
}

\newglossaryentry{attention}{
    name={Attention Mechanism},
    description={A technique that allows models to focus on relevant parts of the input when processing sequences}
}

\newglossaryentry{transformer}{
    name={Transformer},
    description={A neural network architecture based on self-attention mechanisms, eliminating the need for recurrence and convolutions}
}

\newglossaryentry{ResNet}{
    name={Residual Network (ResNet)},
    description={A CNN architecture that uses skip connections to enable training of very deep networks}
}

\newglossaryentry{VGG}{
    name={VGG Network},
    description={A CNN architecture known for its simplicity, using only $3\times3$ convolutional layers with deep networks}
}

\newglossaryentry{AlexNet}{
    name={AlexNet},
    description={A CNN architecture that won the ImageNet competition in 2012, popularizing deep learning in computer vision}
}

\newglossaryentry{BatchSize}{
    name={Batch Size},
    description={The number of samples processed before model parameters are updated}
}

\newglossaryentry{optimizer}{
    name={Optimizer},
    description={An algorithm that updates model parameters based on computed gradients}
}

\newglossaryentry{gradient}{
    name={Gradient},
    description={A vector of partial derivatives representing the direction and rate of steepest increase of a function}
}

\newglossaryentry{maxpool}{
    name={Max Pooling},
    description={A pooling operation that takes the maximum value from each patch of a feature map}
}

\newglossaryentry{avgpool}{
    name={Average Pooling},
    description={A pooling operation that computes the average value from each patch of a feature map}
}

\newglossaryentry{convolution}{
    name={Convolution},
    description={A mathematical operation that combines two functions to produce a third function, fundamental to CNNs}
}

\newglossaryentry{featuremap}{
    name={Feature Map},
    description={The output of a convolutional layer, representing detected features at different spatial locations}
}

\newglossaryentry{receptivefield}{
    name={Receptive Field},
    description={The region of the input space that affects a particular neuron in the network}
}
