\section{线性代数}

\subsection{矩阵运算}

\begin{itemize}
    \item $\mathbf{AB} \neq \mathbf{BA}$（非交换）
    \item $(\mathbf{AB})^\top = \mathbf{B}^\top\mathbf{A}^\top$
    \item $(\mathbf{AB})^{-1} = \mathbf{B}^{-1}\mathbf{A}^{-1}$
    \item $\det(\mathbf{AB}) = \det(\mathbf{A})\det(\mathbf{B})$
\end{itemize}

\section{微积分}

\subsection{常用导数}

\begin{align}
\frac{d}{dx}x^n &= nx^{n-1} \\
\frac{d}{dx}e^x &= e^x \\
\frac{d}{dx}\log(x) &= \frac{1}{x} \\
\frac{d}{dx}\sin(x) &= \cos(x) \\
\frac{d}{dx}\cos(x) &= -\sin(x)
\end{align}

\chapter{术语表}

\begin{itemize}
    \item \textbf{ANN：} 人工神经网络 (Artificial Neural Network)
    \item \textbf{CNN：} 卷积神经网络 (Convolutional Neural Network)
    \item \textbf{RNN：} 循环神经网络 (Recurrent Neural Network)
    \item \textbf{LSTM：} 长短期记忆网络 (Long Short-Term Memory)
    \item \textbf{GAN：} 生成对抗网络 (Generative Adversarial Network)
    \item \textbf{VAE：} 变分自编码器 (Variational Autoencoder)
    \item \textbf{GN：} 图神经网络 (Graph Neural Network)
    \item \textbf{FLOPs：} 浮点运算次数 (Floating Point Operations)
    \item \textbf{GFLOPs：} 吉咖浮点运算 ($10^9$ FLOPs)
    \item \textbf{BatchNorm：} 批归一化 (Batch Normalization)
    \item \textbf{Dropout：} Dropout正则化技术
    \item \textbf{LR：} 学习率 (Learning Rate)
\end{itemize}
