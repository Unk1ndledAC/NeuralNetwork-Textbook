\chapter{Mathematical Reference}

\section{Linear Algebra}

\subsection{Matrix Operations}

\begin{itemize}
    \item $\mathbf{AB} \neq \mathbf{BA}$ (non-commutative)
    \item $(\mathbf{AB})^\top = \mathbf{B}^\top\mathbf{A}^\top$
    \item $(\mathbf{AB})^{-1} = \mathbf{B}^{-1}\mathbf{A}^{-1}$
    \item $\det(\mathbf{AB}) = \det(\mathbf{A})\det(\mathbf{B})$
\end{itemize}

\section{Calculus}

\subsection{Common Derivatives}

\begin{align}
\frac{d}{dx}x^n &= nx^{n-1} \\
\frac{d}{dx}e^x &= e^x \\
\frac{d}{dx}\log(x) &= \frac{1}{x} \\
\frac{d}{dx}\sin(x) &= \cos(x) \\
\frac{d}{dx}\cos(x) &= -\sin(x)
\end{align}

\subsection{Chain Rule}

For $y = f(g(x))$:
\begin{equation}
\frac{dy}{dx} = f'(g(x)) \cdot g'(x)
\end{equation}

\section{Probability}

\begin{align}
P(A|B) &= \frac{P(B|A)P(A)}{P(B)} \\
E[X] &= \sum_x x \cdot P(X=x) \\
\text{Var}(X) &= E[X^2] - (E[X])^2
\end{align}
